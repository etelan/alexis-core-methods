public static int search(int[] nums, int target, int left, int right) {
                
        while (left <= right) {
        
            int index = (int) Math.ceil((left + right) /2f);
                        
            if (nums[index] == target) {
                return index;
            } else if (nums[index] < target) {
                left = index + 1;
            } else if ((nums[index] > target) ){
                right = index - 1;
            }
                        
        }
        
        return -1;
    }
